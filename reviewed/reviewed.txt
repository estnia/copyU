三、剩余高风险 / 影响性能的点（必须优先处理）

P1（高优先）
	1.	主线程阻塞 — QProcess.waitForFinished() 或同步等待
	•	现象/风险：若仍在主线程使用 waitForFinished()（或其它同步等待），会导致 UI 卡顿，影响响应性与用户体验；并在高并发场景下放大。
	•	建议：改为 异步启动 QProcess + 使用 finished 信号或 QTimer 做超时保护，或把等待移入后台 worker（不在 GUI 线程）。
	•	可用改动（patch）示例见下方 代码片段 A（异步 QProcess + QTimer）。
	2.	数据库事务与连接管理不够防护
	•	现象/风险：多处使用 sqlite3.connect() + 显式 commit()/close()，在异常路径可能产生未关闭连接或未回滚事务，导致 DB 锁、数据不一致、资源泄露。
	•	建议：统一封装 DB helper，使用 with sqlite3.connect(...) as conn:，启用 PRAGMA journal_mode=WAL;，设置合理 timeout，并对写入路径做重试（backoff）与最大并发限制。
	•	可用改动示例：代码片段 B（context manager + WAL + timeout）。
	3.	任务队列无上限 / 线程创建策略未统一
	•	现象/风险：存在 task_queue、queue_lock 等结构但未使用限长队列，未来热点路径会导致内存无限增长或大量线程创建。
	•	建议：用 queue.Queue(maxsize=...) 或 collections.deque(maxlen=...)；把 worker 模型改为 QThreadPool + QRunnable，并对入队失败（满队）采取 backpressure 或丢弃策略。
	•	可用改动示例：代码片段 C（QRunnable worker + queue 限长）。

代码片段 A — 异步 QProcess + QTimer 超时保护（替换同步 wait）

# 异步 QProcess 启动 xdotool，非阻塞主线程
from PyQt5.QtCore import QProcess, QTimer

class XDoToolRunner:
    def __init__(self, parent=None, timeout_ms=1000):
        self._proc = QProcess(parent)
        self._proc.setProgram("/usr/bin/xdotool")
        self._timeout_ms = timeout_ms

    def run_async(self, args:list, on_finished=None):
        # args: ["keyup", "ctrl", ...]
        if self._proc.state() != QProcess.NotRunning:
            try:
                self._proc.kill()
            except Exception:
                pass
        self._proc.setArguments(args)

        finished_flag = {'done': False}
        def _on_finished(exitCode, exitStatus):
            finished_flag['done'] = True
            if on_finished:
                on_finished(exitCode, exitStatus)
            # 可读取 stderr/stdout 如需日志：
            # err = bytes(self._proc.readAllStandardError()).decode(errors='ignore')

        self._proc.finished.connect(_on_finished)
        self._proc.start()

        # 超时保护（非阻塞）
        timer = QTimer()
        timer.setSingleShot(True)
        def _on_timeout():
            if not finished_flag['done']:
                try:
                    self._proc.kill()
                except Exception:
                    pass
                # 记录超时日志
                logger.warning("xdotool async timeout, args=%s", args)
        timer.timeout.connect(_on_timeout)
        timer.start(self._timeout_ms)
        return True  # 立即返回

说明：若上层需要等待结果，请把调用方改为监听回调/信号或在 worker 线程中同步等待；不要在 GUI 线程阻塞等待。

import sqlite3
from contextlib import contextmanager

@contextmanager
def sqlite_conn(db_path, timeout=5):
    conn = sqlite3.connect(db_path, timeout=timeout, isolation_level=None)  # autocommit off with explicit transactions
    try:
        conn.execute("PRAGMA journal_mode=WAL;")
        conn.execute("PRAGMA synchronous=NORMAL;")
        yield conn
        # commit/rollback handled by caller via explicit transaction scope, or use conn.commit() here if desired
    finally:
        try:
            conn.close()
        except Exception:
            pass

# 使用示例
with sqlite_conn(self.db_path) as conn:
    cur = conn.cursor()
    try:
        cur.execute("BEGIN")
        cur.execute("INSERT ...", params)
        conn.commit()
    except Exception:
        conn.rollback()
        logger.exception("db write failed")
        raise

代码片段 C — QThreadPool Worker & 限长 Queue

from PyQt5.QtCore import QRunnable, QThreadPool, pyqtSignal, QObject
from queue import Queue, Full

class WorkerSignals(QObject):
    finished = pyqtSignal(object, object)  # (result, exception)

class Worker(QRunnable):
    def __init__(self, fn, *args, **kwargs):
        super().__init__()
        self.fn = fn
        self.args = args
        self.kwargs = kwargs
        self.signals = WorkerSignals()

    def run(self):
        try:
            res = self.fn(*self.args, **self.kwargs)
            self.signals.finished.emit(res, None)
        except Exception as e:
            self.signals.finished.emit(None, e)

# 全局队列（生产者入队）
task_queue = Queue(maxsize=500)

def submit_task(fn, *args, **kwargs):
    try:
        task_queue.put_nowait((fn, args, kwargs))
    except Full:
        logger.warning("task queue full, drop task")
        return False
    return True

# 消费者：把队列任务派发到 QThreadPool
def dispatcher_loop():
    pool = QThreadPool.globalInstance()
    while True:
        fn, args, kwargs = task_queue.get()
        pool.start(Worker(fn, *args, **kwargs))
        task_queue.task_done()
